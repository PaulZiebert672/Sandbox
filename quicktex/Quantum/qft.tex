\begin{array}{l}
  \text{Lagrangian field theory} \\
  S = \displaystyle\int\mathcal{L}\left(\Phi,\Phi_{,\mu}\right)\,\mathrm{d}^4x,\;
  \delta S = 0 \\
  \partial_\mu\left(\dfrac{\partial\mathcal{L}\phantom{{}_{,\mu}}}{\partial\Phi_{,\mu}}\right) -
  \dfrac{\partial\mathcal{L}}{\partial\Phi} = 0 \\

  \text{Hamiltonian field theory} \\
  \pi\left(\vec{x}\right) = \dfrac{\partial\mathcal{L}\phantom{(\vec{x})}}{\partial\dot{\Phi}\left(\vec{x}\right)} \\
  \small{ \displaystyle\int\mathcal{H}\,\mathrm{d}^3x =
  \int\left[\pi\left(\vec{x}\right)\dot{\Phi}\left(\vec{x}\right) -
  \mathcal{L}\right]\,\mathrm{d}^3x } \\
  \mathrm{T}^\mu_{\phantom{\mu}\nu} =
  \dfrac{\partial\mathcal{L}\phantom{{}_{,\mu}}}{\partial\Phi_{,\mu}}\Phi_{,\nu} -
  \mathcal{L}\,\delta^\mu_{\phantom{\mu}\nu} \\
\end{array}
