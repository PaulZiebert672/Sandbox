\begin{array}{l}

  \text{Quantum state evolution} \\
  i \partial_t |\Psi(t)\rangle =
  \hat{\mathcal{H}}(\hat{\mathbf{p}}, \hat{\mathbf{x}}, t) |\Psi(t)\rangle \\
  \left [ \hat{p}_r,\hat{x}_s \right ] = -i\delta_{rs} \\
  \left [ \hat{p}_r,\hat{p}_s \right ] = \left [ \hat{x}_r,\hat{x}_s \right ] = 0 \\

  \text{Representation}\;\mathcal{X},\;
  \hat{\mathcal{X}}\vert{n}\rangle = \xi\vert{n}\rangle \\
  \begin{array}{cc}
    \vert{n}\rangle\langle{n}\vert = \hat{\mathcal{P}}_{n}, &
    \sum_{n}\mathcal{\hat{P}}_{n} = \hat{\mathcal{I}}
  \end{array} \\
  \psi(\xi,t) = \langle{n}\vert\Psi(t) \rangle \\
  
  \text{Observable}\;\Omega\; \text{with} \;
  \langle\Psi\vert\Psi\rangle = 1 \\
  \langle\Omega\rangle = \langle\Psi(t)\vert\hat{\Omega}\vert\Psi(t)\rangle \\
  \dot{\langle\Omega\rangle} = \langle\partial_{t}\hat{\Omega}\rangle +
  i\langle\big[ \hat{\mathcal{H}},\hat{\Omega} \big]\rangle \\
  
  \text{Stationary case:} \quad \partial_{t}\hat{\mathcal{H}} = 0 \\
  \hat{\mathcal{H}} \vert\psi\rangle = \mathscr{E} \vert\psi\rangle \\
  \vert\Psi(t)\rangle = \vert\psi\rangle \exp(-i\mathscr{E}t)
\end{array}
