\begin{array}{l}

\text{Macroscopic electrodynamics} \\

\begin{cases}
\nabla\cdot\mathbf{D} & = 4\pi\rho \\
\nabla\times\mathbf{H} - \dfrac{1}{c}\dfrac{\partial\mathbf{D}}{\partial t} & = \dfrac{4\pi}{c}\mathbf{j}
\end{cases} \\

\begin{cases}
\nabla\cdot\mathbf{B} & = 0 \\
\nabla\times\mathbf{E} + \dfrac{1}{c}\dfrac{\partial\mathbf{B}}{\partial t} & = 0
\end{cases} \\

\text{Constituitive relations} \\
\begin{array}{cc}
\mathbf{D}=\mathbf{D}(\mathbf{E},\mathbf{B}) &
\mathbf{H}=\mathbf{H}(\mathbf{E},\mathbf{B}) \\
\mathbf{j}=\mathbf{j}(\mathbf{E},\mathbf{B})
\end{array} \\

\text{Boundary conditions} \\
\begin{array}{cc}
\scriptsize { (\mathbf{D}_2 - \mathbf{D}_1)\cdot\mathbf{n} = 4\pi\sigma } &
\scriptsize { (\mathbf{B}_2 - \mathbf{B}_1)\cdot\mathbf{n} = 0 } \\
\scriptsize { \mathbf{n}\times(\mathbf{H}_2 - \mathbf{H}_1) = \dfrac{4\pi}{c}\mathbf{K} } &
\scriptsize { \mathbf{n}\times(\mathbf{E}_2 - \mathbf{E}_1) = 0 }
\end{array} \\

\end{array}
