\begin{array}{l}
  \text{Set operations} \\
  \complement A = \{x\vert x\notin A\} \\
  A\setminus B = \{x\vert x\in A\text{ and }x\notin B\} \\
  A\cap B = \{x\vert x\in A\text{ and }x\in B\} \\
  A\cup B = \{x\vert x\in A\text{ or }x\in B\} \\
  2^A = \{B\vert B\subset A\} \\

  A\cap B\subset A,\quad A\cup B\supset A \\

  \text{De Morgan laws} \\
  \complement(A\cup B) = \complement A\cap\complement B \\
  \complement(A\cap B) = \complement A\cup\complement B \\

  \text{Distributive laws} \\
  \small{ A\cap(B_1\cup B_2) = (A\cap B_1)\cup(A\cap B_2) } \\
  \small{ A\cup(B_1\cap B_2) = (A\cup B_1)\cap(A\cup B_2) } \\

  \text{Relative complement} \\
  A\setminus B = A\cap\complement B \\
  A\triangle B = (A\setminus B)\cup(B\setminus A) \\

  \text{Indexed collections} \\
  \bigcap\limits_{\iota\in \mathscr{C}}X_\iota =
  \big\{x\big\vert\forall\iota\in \mathscr{C},x\in X_\iota\big\} \\
  \bigcup\limits_{\iota\in \mathscr{C}}X_\iota =
  \big\{x\big\vert\exists\iota\in \mathscr{C}:x\in X_\iota\big\} \\
\end{array}
