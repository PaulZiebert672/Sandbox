\begin{array}{l}
  \sigma\text{-algebra}\;\mathfrak{A}\subset\,2^X \\
  \begin{cases}
    X\in\mathfrak{A} \\
    \{A_j\}_{j=1}^\infty \subset\mathfrak{A}\Rightarrow
    \bigcup\limits_{j=1}^\infty A_j \in\mathfrak{A} \\
    \{A_1,A_2\}\subset\mathfrak{A}\Rightarrow
    \,A_1\backslash A_2\in\mathfrak{A}
  \end{cases} \\

  \text{De Morgan laws} \\
  \{A_j\}_{j=1}^\infty \subset\mathfrak{A}\Rightarrow
  \bigcap\limits_{j=1}^\infty A_j \in\mathfrak{A} \\

  \sigma\text{-algebra generated by }\mathfrak{S}\subset\,2^X \\
  \sigma(\mathfrak{S}) = \left\{
  \bigcap\limits_{\mathfrak{S}\subset\mathfrak{A}}\mathfrak{A}\Big\vert
  \small{ \mathfrak{A}\;\sigma\text{-algebra on }X }
  \right\} \\

  \text{Measure on a }\sigma\text{-algebra }\mathfrak{A} \\
  \mu:\mathfrak{A}\rightarrow[0,\infty] \\
  \begin{cases}
    \mu(\emptyset) = 0 \\
    {\small \mu\left(\bigcup\limits_{j=1}^\infty A_j\right) =
    \sum\limits_{j=1}^\infty \mu\left(A_j\right), } \\
    \qquad{\small \forall A_k, k\ne j: A_j\cap A_k = \emptyset }
  \end{cases} \\

  \text{Continuity from above} \\
  \begin{matrix}
    \small{ \mu(A_k)<\infty } \\
    \small{ A_n\searrow A }
  \end{matrix}\bigg\}\Rightarrow
  \displaystyle\lim_{n\to\infty}\mu(A_n)=\mu(A) \\
\end{array}
