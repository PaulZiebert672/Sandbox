\begin{array}{l}
  \text{Axiom of choice }
  \mathscr{C}=\left\{A\ne\emptyset\right\} \\
  \exists f:\mathscr{C}\to\!\!\!\!\bigcup\limits_{A\in\mathscr{C}}\!\!A\big\vert
  f(A)\in A,\forall A\in\mathscr{C} \\

  \text{Partial ordering }
  \forall x,y,z\in\Omega \\
  \begin{cases}
    x\prec x \\
    x\prec y,y\prec x\Rightarrow x = y \\
    x\prec y,y\prec z\Rightarrow x\prec z
  \end{cases} \\

  \text{Chain }C\subset(\Omega,\prec) \\
  x\prec y\text{ or }y\prec x,\forall x,y\in C \\

  \text{Upper bound for }A\subset(\Omega,\prec) \\
  u\in\Omega:x\prec u,\forall x\in A \\

  \text{Maximal element }m\in(\Omega,\prec) \\
  x\in\Omega,m\prec x\Rightarrow x = m \\

  \text{Equivalence }
  \forall x,y,z\in\Omega \\
  \begin{cases}
    x\sim x \\
    x\sim y\Rightarrow y\sim x \\
    x\sim y,y\sim z\Rightarrow x\sim z
  \end{cases} \\
\end{array}
