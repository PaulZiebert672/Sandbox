\begin{array}{l}
  \text{Field axioms }
  \forall x,y,z\in\mathbb{R} \\

  \begin{cases}
    x + y = y + x \\
    x + (y + z) = (x + y) + z \\
    \exists 0\in\mathbb{R}: x + 0 = x \\
    \forall x \exists y\in\mathbb{R}: x + y = 0
  \end{cases} \\

  \begin{cases}
    x y = y x \\
    x (y z) = (x y) z \\
    \exists 1\in\mathbb{R},1\ne 0: x 1 = x \\
    \forall x \exists y\in\mathbb{R}: x y = 1
  \end{cases} \\

  \phantom{\{}\:(x + y) z = x z + y z \\

  \text{Order axioms} \\
  \begin{cases}
    x < y,y < z \Rightarrow x < z \\
    x < y \Rightarrow x + z < y + z \\
    x < y,z > 0 \Rightarrow x z < y z \\
    x < y \text{ or } x = y \text{ or } x > y
  \end{cases} \\

  \text{Completeness axiom} \\
  \phantom{\{}\:\left\{A\subset\mathbb{R}\vert
  \exists b\in\mathbb{R}:x < b\;\forall x\in A\right\} \\
  \qquad\Rightarrow\sup A\in\mathbb{R} \\
\end{array}
