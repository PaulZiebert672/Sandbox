\begin{array}{l}
  \text{Topology}\;\mathscr{T}\subset\,2^X \\
  \begin{cases}
    \emptyset\in\mathscr{T}, X\in\mathscr{T} \\
    \{U_\alpha\}_{\alpha\in A}\subset\mathscr{T}\Rightarrow
    \bigcup\limits_{\alpha\in A}U_\alpha\in\mathscr{T} \\
    \{U_1,U_2\}\subset\mathscr{T}\Rightarrow
    \,U_1\cap U_2\in\mathscr{T}
  \end{cases} \\

  \text{Base}\;\mathscr{B}\subset\mathscr{T} \\
  \forall U\in\mathscr{T}:\;
  U = \bigcup\limits_{B\in\mathscr{B}}B \\

  \text{Open neighborhood for}\;p\in X, \\
  \quad U_p\in\mathscr{T}\vert p\in U_p \\

  p\in S\text{ is an interior point of } S\subset X \\
  \quad\Leftarrow \exists U_p\subset S \\

  S^\circ = \{p\in S\vert
  p\small{ \text{ interior point of } }S\} \\
  \complement\overline{S} = \{p\notin S\vert
  p\small{ \text{ interior point of } }\complement{S}\} \\

  \text{Hausdorff:}\quad\forall\{p_1,p_2\}\subset X \\
  \exists U_j\in\mathscr{T}: p_j\in U_j
  \vert U_1\cap U_2 = \emptyset \\
\end{array}
