\begin{array}{l}
  \text{Metric} \\
  \mathrm{d}s^2 = \mathrm{d}\theta^2 + \sin^2\theta\mathrm{d}\phi^2 \\

  \text{Christoffel symbols} \\
  \small{ \Gamma^\theta = \begin{bmatrix}0 & 0 \\ 0 & \!\!\!\!-\frac{\sin 2\theta}{2}\end{bmatrix},\,
  \Gamma^\phi = \begin{bmatrix}0 & \cot\theta \\ \cot\theta & 0\end{bmatrix} } \\

  \text{Killing vectors} \\
  \begin{array}{lll}
    \xi^{(1)} & = & \sin\phi\partial_\theta + \cot\theta\cos\phi\partial_\phi \\
    \xi^{(2)} & = & \cot\theta\sin\phi\partial_\phi - \cos\phi\partial_\theta \\
    \xi^{(3)} & = & \partial_\phi
  \end{array} \\
\end{array}
