\begin{array}{l}
  \text{Lie algebra representation} \\
  \varrho:\mathfrak{g}=\left(\mathsf{L},[.,.]\right)\,\rlap{\:\:{}^{\sim}}\longrightarrow\text{End}(\mathsf{V}),\;
  \text{s.t.} \\
  \left[a,b\right]\longmapsto\left[\varrho(a),\varrho(b)\right] \\

  \text{Adjoint representation} \\
  \varrho:\mathsf{L}\,\rlap{\,{}^{\sim}}\rightarrow\text{End}(\mathsf{L}),\;
  a\mapsto\text{ad}(a) = [a,.] \\

  \text{Killing form}\quad
  K_\varrho:\mathsf{L}\times\mathsf{L}\,\rlap{\,{}^{\sim}}\rightarrow\mathbb{C} \\
  a,b\mapsto\text{tr}\left(\varrho(a)\varrho(b)\right) \\

  \text{Lie group representation} \\
  \mathscr{R}:\left(G,\circ\right)\,\longrightarrow{GL}(\mathsf{V}),\;
  \text{s.t.} \\
  \mathscr{R}(g_1)\mathscr{R}(g_2) = \mathscr{R}(g_1\circ g_2) \\  

  \text{Adjoint representation} \\
  \text{Ad}_g:G\,\longrightarrow G,\;
  h\mapsto ghg^{-1} \\
  \text{Ad}_{g^\ast}:\mathsf{T}_eG\,\rlap{\:\:{}^{\sim}}\longrightarrow\mathsf{T}_eG\,\cong\mathsf{L}(G) \\
  \text{Ad}:G\,\longrightarrow GL(\mathsf{T}_eG),\;g\mapsto\text{Ad}_{g^\ast} \\
\end{array}
