\begin{array}{l}
  \text{Change of variables}\;
  g:\left[a,b\right]\rightarrow\mathbb{R} \\
  \displaystyle\int\limits_{g(a)}^{g(b)}\!\!\!\!f(y)\,\mathrm{d}y =
  \displaystyle\int\limits_{a}^{b}\!\!f\left(g(x)\right)g'(x)\,\mathrm{d}x \\
  \omega = f(y)\mathrm{d}y,\;
  g^{*}\omega = \left(f\circ g\right)\mathrm{d}g \\

  \text{Pullback}\quad
  h:\,\mathbb{R}^n\supset U\rightarrow\,\mathbb{R}^m \\
  h^{*}:\Lambda^k\left(\mathbb{R}^m\right)\rightarrow\Lambda^k\left(U\right) \\
  \begin{cases}
    h^{\ast}f & = & f{\circ}h \\
    h^{\ast}\mathrm{d}x_i & = & \dfrac{\partial h_i}{\partial x_j}\mathrm{d}x_j \\
    h^{\ast}\left(\omega\wedge\eta\right) & = & h^{\ast}\omega\wedge h^{\ast}\eta
  \end{cases} \\

  h^{\ast}\mathrm{d}x_I = \left\vert\dfrac{\partial h_I}{\partial x_J}\right\vert
  \mathrm{d}u_J,\;\vert J\vert = k \\
\end{array}
