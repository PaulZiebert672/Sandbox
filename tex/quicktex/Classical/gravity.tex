\begin{array}{l}
  \text{Einstein equations} \\
  \mathrm{R}_{\mu\nu} -
  \frac{1}{2}\mathrm{g}_{\mu\nu}
  \mathrm{g}^{\sigma\rho}\mathrm{R}_{\rho\sigma}\, =
  8\pi G_{\mathtt{N}}\mathrm{T}_{\mu\nu} \\
  \mathrm{R}_{\mu\nu} =
  8\pi G_{\mathtt{N}}\left(\mathrm{T}_{\mu\nu} -
  \frac{1}{2}\mathrm{g}_{\mu\nu}
  \mathrm{g}^{\sigma\rho}\mathrm{T}_{\rho\sigma}\right) \\  

  \text{Ricci tensor} \\
  \begin{array}{lll}
    \mathrm{R}_{\mu\nu} & = &
    \partial_\sigma\Gamma^\sigma_{\mu\nu} -
    \partial_\nu\Gamma^\sigma_{\mu\sigma} \\
    & + & \Gamma^\sigma_{\rho\sigma}\Gamma^\rho_{\mu\nu} -
    \Gamma^\sigma_{\rho\nu}\Gamma^\rho_{\mu\sigma}
  \end{array} \\

  \text{Contracted Bianchi identity} \\
  \left(\mathrm{R}^{\mu\nu} -
  \frac{1}{2}\mathrm{g}^{\mu\nu}
    \mathrm{g}^{\sigma\rho}\mathrm{R}_{\rho\sigma}\right)_{;\mu} = 0 \\

  \text{Christoffel symbols} \\
  \Gamma^\sigma_{\mu\nu} = \frac{1}{2}\mathrm{g}^{\sigma\rho}
  \left(\partial_\mu\mathrm{g}_{\rho\nu} +
  \partial_\nu\mathrm{g}_{\mu\rho} -
  \partial_\rho\mathrm{g}_{\mu\nu}\right) \\

  \Gamma^\sigma_{\mu\sigma} = \partial_\mu\log\sqrt{-\mathrm{g}} \\

  \text{Geodesic equation} \\
  \ddot{x}^\sigma +
  \Gamma^\sigma_{\mu\nu}\dot{x}^\mu\dot{x}^\nu = 0 \\

  \text{Schwarzschild metric}\quad m=G_{\mathtt{N}}M \\
  \mathrm{d}s^2 = \left(1 - \frac{2m}{r}\right)\mathrm{d}t^2 -
  \dfrac{\mathrm{d}r^2}{1 - \frac{2m}{r}} \\
  \qquad- r^2\left(\mathrm{d}\theta^2 + \sin^2\theta\mathrm{d}\phi^2\right) \\ 

  \text{Robertson-Wanker metric} \\
  \mathrm{d}s^2 = \mathrm{d}t^2 - a^2(t)\mathrm{d}\varsigma^2 \\
  \small{ \mathrm{d}\varsigma^2 = \dfrac{\mathrm{d}r^2}{1-\kappa r^2}+r^2\left(\mathrm{d}\theta^2 +
  \sin^2\theta \mathrm{d}\phi^2\right) } \\
\end{array}
