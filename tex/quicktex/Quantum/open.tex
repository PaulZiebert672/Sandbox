\begin{array}{l}
  \text{Composite systems} \\
  \mathfrak{H} = \,\mathfrak{H}_S\otimes \mathfrak{H}_E \\
  \langle X\otimes Y\rangle\ne
  \langle X\otimes\mathbb{I}_E\rangle\langle\mathbb{I}_S\otimes Y\rangle \\

  \text{Reduced density operator} \\
  {\small \mathrm{tr}\left(\rho_S X\right) = 
  \mathrm{tr}\left(\rho(X\otimes\mathbb{I}_E)\right),\:
  \forall X\in\,\mathcal{A}_S } \\

  \text{Partial trace} \\
  \mathrm{tr}_E:\rho\mapsto\rho_S = 
  \sum_n\langle n\vert\rho\vert n\rangle_E \\

  \text{Classical correlation} \\
  \rho = \sum_x p_x \rho_S\otimes\rho_E \\
\end{array}
