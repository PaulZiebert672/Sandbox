\begin{array}{l}
  \text{Equation of state}\quad p = p(v, \theta) \\
  \mathrm{d}p = \Big(\dfrac{\partial{p}}{\partial{v}}\Big)_{\theta}\mathrm{d}v +
  \Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_{v}\mathrm{d}\theta \\

  \text{For}\quad p = \text{const.} \\
  0 = \Big(\dfrac{\partial{p}}{\partial{v}}\Big)_{\theta}
  \Big(\dfrac{\partial{v}}{\partial{\theta}}\Big)_{p} +
  \Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_{v} \\
  \mathcal{K}_{\theta}\alpha_{p} = p\beta_{v} \\

  \text{First law}\quad \delta{q} = \mathrm{d}u + p\mathrm{d}v \\
  \mathrm{d}q = c_v\mathrm{d}\theta +
  \Big[\Big(\dfrac{\partial{u}}{\partial{v}}\Big)_{\theta} +
  p\Big]\mathrm{d}v \\

  \text{Second law}\quad \delta{q} = \theta\mathrm{d}\varsigma \\
  \dfrac{\partial}{\partial{\theta}}\Bigg\vert_{v}
  \dfrac{c_v}{\theta} =
  \dfrac{\partial}{\partial{v}}\Bigg\vert_{\theta}
  \Bigg[\dfrac{1}{\theta}
  \Big(\dfrac{\partial{u}}{\partial{v}}\Big)_{\theta} +
  \dfrac{p}{\theta}\Bigg] \\

  \begin{array}{lll}
    \Big(\dfrac{\partial{u}}{\partial{v}}\Big)_{\theta} & = &
    \theta\Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_{v} - p \\
    & = & \theta\mathcal{K}_\theta\alpha_{p} - p
  \end{array}
\end{array}
