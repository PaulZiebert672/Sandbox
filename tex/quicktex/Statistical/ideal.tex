\begin{array}{l}
  \text{Classical ideal gas} \\
  \begin{cases}
    pv = \theta \\
    u = c_{v}\theta & \text{with} &
    \frac{\partial{c_v}}{\partial\theta} = 0
  \end{cases} \\

  \text{Mayer's relation} \\
  c_{p} = \Big(\dfrac{\partial{h}}{\partial{\theta}}\Big)_{p} = c_{v} + 1 \\

  \text {Adiabatic process} \\
  \delta{q} = \mathrm{d}u + p\mathrm{d}v = 0 \\
  pv^\gamma = \text{const.} \quad
  \gamma = \dfrac{c_{p}}{c_{v}} \\

  \text{Isothermal bulk modulus} \\
  \mathcal{K}_{\theta} = p \\

  \text{Isobaric thermal expansion} \\
  \alpha_{p} = \theta^{-1}
\end{array}
