\begin{array}{l}
  \text{Perfect fluid} \\
  \mathrm{T}_{\mu\nu} =
  (p + \varrho)u_{\mu}u_{\nu} - \eta_{\mu\nu}p \\

  \mathrm{T}_{\mu\nu} \doteq
  \begin{bmatrix}
    \varrho & 0 \\
    0 & p\delta_{ij}
  \end{bmatrix} \\

  \text{Electromagnetic field} \\
  \small{ 4\pi\mathrm{T}_{\mu\nu} =
  \,\mathrm{F}_{\mu\sigma}\eta^{\sigma\rho}\mathrm{F}_{\rho\nu} +
  \frac{1}{4}\eta_{\mu\nu}\mathrm{F}_{\sigma\rho}\mathrm{F}^{\sigma\rho} } \\

  \small{ 4\pi\mathrm{T}_{\mu\nu} =
  \begin{bmatrix}
    \frac{1}{2}(E^2+B^2) &
    -\epsilon_{ijk}E_{j}B_{k} \\
    -\epsilon_{ijk}E_{j}B_{k} &
    -\sigma_{ij}
  \end{bmatrix} } \\

  \small{ \sigma_{ij} = E_{i}E_{j} + B_{i}B_{j} -
  \frac{1}{2}\delta_{ij}(E^2 + B^2) } \\
\end{array}
