\begin{array}{l}
  \text{Free particle} \\
  S = -m\displaystyle\int\sqrt{\mathrm{d}x^\mu\mathrm{d}x_\mu}\\
  S = -m\displaystyle\int\sqrt{\mathrm{g}_{\mu\nu}\dot{q}^\mu\dot{q}^\nu}\,\mathrm{d}s \\

  \text{Free complex scalar field} \\
  S = \displaystyle\int\left(\partial^\mu\phi^*\partial_\mu\phi - m^2\phi^*\phi\right)\,\mathrm{d}^4x \\

  \text{Free spinor field} \\
  S = \displaystyle\int\bar{\psi}\left(i\gamma^\mu\partial_\mu - m\right)\psi\,\mathrm{d}^4x \\
  \quad\bar{\psi} = \psi^{\dagger}\gamma^0 \\

  \text{Free massless vector field} \\
  S = -\dfrac{1}{16\pi}\displaystyle\int\mathrm{F}_{\mu\nu}\mathrm{F}^{\mu\nu}\,\mathrm{d}^4x \\
  \quad\mathrm{F}_{\mu\nu} = \partial_\mu A_\nu - \partial_\nu A_\mu \\

  \text{Gravitational field} \\
  S = -\dfrac{1}{16\pi G_{\mathtt{N}}}
  \displaystyle\int\mathrm{R}\sqrt{-\mathrm{g}}\,\mathrm{d}^4x \\
  \quad\mathrm{R}_{\rho\nu} = \,\mathrm{R}^\sigma_{\phantom{\sigma}\rho\sigma\nu},\quad
  \mathrm{R} = \,\mathrm{g}^{\nu\rho}\mathrm{R}_{\rho\nu} \\
\end{array}
