\begin{array}{l}
  \text{Commutator of derivatives} \\
  \small{ X^\rho_{\phantom{\rho};\nu\mu} \!- X^\rho_{\phantom{\rho};\mu\nu} \!=
  \,\mathrm{T}^{\sigma}_{\phantom{\sigma}\mu\nu}X^\rho_{\phantom{\rho},\sigma} \!+
  \,\mathrm{R}^{\rho}_{\phantom{\rho}\sigma\mu\nu}X^\sigma } \\
  \small{ \omega_{\rho;\nu\mu} - \omega_{\rho;\mu\nu} =
  \,\mathrm{T}^{\sigma}_{\phantom{\sigma}\mu\nu}\omega_{\rho,\sigma} -
  \,\mathrm{R}^{\sigma}_{\phantom{\sigma}\rho\mu\nu}\omega_\sigma } \\

  \text{Torsion}\quad
  \mathrm{T}^{\lambda}_{\phantom{\lambda}\mu\nu} =
  \Gamma^\lambda_{\phantom{\lambda}\mu\nu} - \Gamma^\lambda_{\phantom{\lambda}\nu\mu} \\

  \text{Curvature with}\quad\mathrm{T}^{\lambda}_{\phantom{\sigma}\mu\nu} = 0 \\
  \begin{array}{lll}
    \mathrm{R}^{\rho}_{\phantom{\rho}\sigma\mu\nu} & = &
    \Gamma^\rho_{\phantom{\rho}\sigma\nu,\mu} -
    \Gamma^\rho_{\phantom{\rho}\sigma\mu,\nu} \\
    & + & \Gamma^{\rho}_{\phantom{\rho}\lambda\mu}\Gamma^\lambda_{\phantom{\lambda}\sigma\nu} -
    \Gamma^{\rho}_{\phantom{\rho}\lambda\nu}\Gamma^\lambda_{\phantom{\lambda}\sigma\mu} 
  \end{array} \\
  \frac{n^2(n^2-1)}{12}\;\text{independent components} \\

  \text{Parallel transport} \\
  \delta X^\rho = \,\mathrm{R}^{\rho}_{\phantom{\rho}\sigma\mu\nu}\delta u^\mu\delta v^\nu X^\sigma \\

  \text{Differential Bianchi identity} \\
  \mathrm{R}^{\rho}_{\phantom{\rho}\sigma[\mu\nu;\lambda]} = 0 \\

  \text{In geodesic coordinates} \\
  \mathrm{R}_{\rho\sigma\mu\nu}\doteq
  \frac{1}{2}\footnotesize{ \left(
  \mathrm{g}_{\nu\rho,\sigma\mu} -
  \mathrm{g}_{\nu\sigma,\rho\mu} +
  \mathrm{g}_{\mu\sigma,\rho\nu} -
  \mathrm{g}_{\mu\rho,\sigma\nu}
  \right) } \\

  \text{For maximally symmetric space} \\
  \mathrm{R_{\rho\sigma\mu\nu}}\doteq
  \frac{\mathrm{R}}{n(n-1)}
  \small{ \left(\mathrm{g}_{\rho\mu}\mathrm{g}_{\sigma\nu} -
  \mathrm{g}_{\rho\nu}\mathrm{g}_{\sigma\mu}\right) } \\
\end{array}
