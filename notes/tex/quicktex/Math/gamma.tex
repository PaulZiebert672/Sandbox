\begin{array}{l}
  \text{Gamma function} \\
  \Gamma(z) = \displaystyle\int_0^\infty
  t^{z-1}\mathrm{e}^{-t}\,\mathrm{d}t \\
  \small{ \Gamma(z+1) = z\Gamma(z),\;\Gamma(n+1)\Big\vert_{n\in\mathbb{N}} = n! } \\

  \sqrt{\pi}\,\Gamma(2z) = 2^{2z-1}\Gamma(z)\Gamma(z+\frac{1}{2}) \\
  \Gamma(z)\Gamma(1-z) =\dfrac{\pi}{\sin\pi z} \\

  \dfrac{1}{\Gamma(z)} = z\mathrm{e}^{\gamma z}\displaystyle\lim_{n\rightarrow\infty}
  \prod_{k=1}^n\left(1+\dfrac{z}{k}\right)\mathrm{e}^{-\frac{z}{k}} \\
  \small{ \;\gamma = \displaystyle\lim_{n\rightarrow\infty}
  \left(\sum_{k=1}^n\dfrac{1}{k} - \log n\right)
  \approx 0.577 } \\

  \text{The measure of balls} \\
  \lambda\left(\mathsf{B}(x\in\mathbb{R}^n;1)\right) =
  \dfrac{\pi^{\frac{n}{2}}}{\Gamma\left(\frac{n}{2} + 1\right)} \\

  \text{Stirling formula} \\
  \Gamma(z+1)\,\rlap{{}^{\!\footnotesize{{}^{z\to\infty}}}}\to
  \sqrt{2\pi}z^{z+\frac{1}{2}}\mathrm{e}^{-z} \\

  \text{Beta function} \\
  \displaystyle\int_0^1 t^{x-1}(1-t)^{y-1}\,\mathrm{d}t =
  \dfrac{\Gamma(x)\Gamma(y)}{\Gamma(x+y)} \\
\end{array}
