\begin{array}{l}
  \text{Differential forms} \\
  \mathrm{d}f = \dfrac{\partial f}{\partial x^i}\,\mathrm{d}x^i \\
  \mathrm{d}\left(f\omega\right) = \mathrm{d}f\wedge\omega + f\mathrm{d}\omega \\
  \mathrm{d}\left(\phi\wedge\omega\right) = \mathrm{d}\phi\wedge\omega + (-1)^k\phi\wedge\mathrm{d}\omega \\
  \mathrm{d}^2\omega = 0 \\

  \text{Integration} \\
  \omega = f(x)\,\mathrm{d}x^1\wedge\cdots\wedge\mathrm{d}x^k \\
  \displaystyle\int_U\omega\equiv\int_U f(x)\,\mathrm{d}x^1\ldots\mathrm{d}x^k \\

  \text{Stokes theorem} \\
  \begin{array}{lll}
  \displaystyle\int_\Omega\mathrm{d}\omega &
  = & \displaystyle\oint_{\partial\Omega}\omega
  \end{array} \\
\end{array}
