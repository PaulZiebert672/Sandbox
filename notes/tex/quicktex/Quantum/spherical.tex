\begin{array}{l}

\text{Spherical harmonics} \\

\quad\ell = 0 \\
\mathrm{Y}_{0}^0(\theta,\phi) = \sqrt{\frac{1}{4\pi}} \\

\quad\ell = 1 \\
\mathrm{Y}_{1}^{\pm 1}(\theta,\phi) = \mp \sqrt{\frac{3}{8\pi}} \sin\theta \, \mathrm{e}^{\pm \mathrm{i}\phi} \\
\mathrm{Y}_{1}^0(\theta,\phi) = \sqrt{\frac{3}{4\pi}} \cos\theta \\

\quad\ell = 2 \\
\mathrm{Y}_{2}^{\pm 2}(\theta,\phi) = \pm \sqrt{\frac{15}{32\pi}} \sin^2\theta \, \mathrm{e}^{\pm 2\mathrm{i}\phi} \\
\mathrm{Y}_{2}^{\pm 1}(\theta,\phi) = \mp \sqrt{\frac{15}{8\pi}} \sin\theta\cos\theta \, \mathrm{e}^{\pm \mathrm{i}\phi} \\
\mathrm{Y}_{2}^0(\theta,\phi) = \sqrt{\frac{5}{16\pi}} (3\cos^2\theta - 1) \\

\quad\ell = 3 \\
\mathrm{Y}_{3}^{\pm 3}(\theta,\phi) = \mp \sqrt{\frac{35}{64\pi}} \sin^3\theta \, \mathrm{e}^{\pm 3\mathrm{i}\phi} \\
\mathrm{Y}_{3}^{\pm 2}(\theta,\phi) = \pm \sqrt{\frac{105}{32\pi}} \, \small{ \sin^2\theta\cos\theta \, \mathrm{e}^{\pm 2\mathrm{i}\phi} } \\
\mathrm{Y}_{3}^{\pm 1}(\theta,\phi) = \mp \sqrt{\frac{21}{64\pi}} \sin\theta \cdot \\
  \phantom{Y_{3}^{\pm 1}(\theta,\phi) = \mp} \cdot(5\cos^2\theta - 1) \, \mathrm{e}^{\pm \mathrm{i}\phi} \\
\mathrm{Y}_{3}^0(\theta,\phi) = \sqrt{\frac{7}{16\pi}} \small{ (5\cos^3\theta - 3\cos\theta) } \\

\end{array}
