\begin{array}{l}
  \text{Density matrix} \\
  i\partial_{t}\hat\rho = [\hat{\mathcal{H}},\hat\rho] \\

  \text{Observable}\ \Omega\ \text{with}\ \mathrm{tr}(\hat\rho) = 1 \\
  \langle\Omega\rangle = \mathrm{tr}(\hat{\rho}\hat{\Omega}) \\

  \text{Thermal equilibrium} \\
  \hat\rho = \dfrac{\mathrm{e}^{-\beta\hat{\mathcal{H}}}}{\mathcal{Z}(\beta)},\quad
  \mathcal{Z}(\beta) = \mathrm{tr}(\mathrm{e}^{-\beta\hat{\mathcal{H}}}) \\

  {\small \mathscr{E}\text{-representation},\;\hat{\mathcal{H}}\vert n\rangle = \,\mathscr{E}_{n}\vert n\rangle } \\
  \langle n\vert\hat{\rho}\vert m\rangle = \,\mathcal{Z}^{-1}(\beta)\mathrm{e}^{-\beta\mathscr{E}_n}\delta_{nm} \\
  \mathcal{Z}(\beta) = \sum_n{\mathrm{e}^{-\beta\mathscr{E}_n}}
  = \,\mathrm{e}^{-\beta\mathcal{F}} \\
  \langle\mathscr{E}\rangle = -\partial_\beta\log\mathcal{Z}(\beta) \\  
  \langle\Delta\mathscr{E}^2\rangle = \partial^2_\beta\log\mathcal{Z}(\beta) \\
\end{array}
