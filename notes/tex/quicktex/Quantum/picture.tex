\begin{array}{l}
  \quad\mathtt{A}.\,\mathrm{Schr\ddot{o}dinger}\text{'s picture} \\
  i\partial_{t}\vert\psi(t)\rangle = \mathcal{H}\vert\psi(t)\rangle \\
  \text{State}\,\vert\psi(t)\rangle = \vert\psi(0)\rangle \mathrm{e}^{-i\mathcal{H}t} \\

  \text{Observable}\,\langle\phi_n(t)\vert\Omega\vert\phi_m(t)\rangle \\
  \dot\Omega = \partial_{t}\Omega + i[\mathcal{H}, \Omega] \\

  \quad\mathtt{B}.\,\text{Heisenberg's picture} \\
  \text{State}\,\vert\psi\rangle \equiv \vert\psi(0)\rangle \\

  \text{Observable} \,\langle\phi_n\vert\Omega(t)\vert\phi_m\rangle \\
  \Omega(t) = \'\mathrm{e}^{i\mathcal{H}t} \Omega(0) \mathrm{e}^{-i\mathcal{H}t} \\
  \dot\Omega(t) = \partial_{t}\Omega(t) + i[\mathcal{H}, \Omega(t)] \\
\end{array}
