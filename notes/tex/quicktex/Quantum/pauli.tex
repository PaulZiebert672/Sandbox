\begin{array}{l}
  \text{Pauli matrices}\quad
  \sigma_x, \sigma_y, \sigma_z \\
  \begin{pmatrix}
    0 & \phantom{-}1 \\ 1 & \phantom{-}0
  \end{pmatrix},
  \begin{pmatrix}
    0 & -i \\ i & \phantom{-}0
  \end{pmatrix},
  \begin{pmatrix}
    1 & \phantom{-}0 \\ 0 & -1
  \end{pmatrix} \\

  {\small \left[\sigma_i,\sigma_j\right] =
  2i\varepsilon_{ijk}\sigma_k,\;
  \left\{\sigma_i,\sigma_j\right\} =
  2\delta_{ij}\sigma_0 } \\

  \text{Properties} \\
  \sigma_i^{\dagger} = \sigma_i^{-1} = \sigma_i, \quad
  \sigma_i^2 = \sigma_0 \\
  \det{\sigma_i} = -1 \\

  \text{Relations} \\
  \small{ (\vec{a},\vec{\sigma})(\vec{b},\vec{\sigma}) =
  (\vec{a},\vec{b})\sigma_0 + i([\vec{a}\times\vec{b}],\vec{\sigma}) } \\

  \mathrm{e}^{i(\hat{n},\vec{\sigma})\phi} =
  \,\sigma_0\cos\phi + i\left(\hat{n},\vec{\sigma}\right)\sin\phi \\

  \mathrm{e}^{i(\hat{n},\vec{\sigma})\phi}\vec{\sigma}
  \mathrm{e}^{-i(\hat{n},\vec{\sigma})\phi} = \\
  \qquad\vec{\sigma}\cos 2\phi + [\hat{n}\times\vec{\sigma}]\sin 2\phi + \\
  \qquad\hat{n}(\hat{n},\vec{\sigma})\left(1 - \cos 2\phi\right) \\
\end{array}
