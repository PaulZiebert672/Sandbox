\begin{array}{l}
  \text{Dirac equation} \\
  \left(i\gamma^\mu\partial_\mu - m\right)\psi = 0 \\
  \left\{\gamma^{\mu},\gamma^{\nu}\right\}=2\eta^{\mu\nu}\mathbb{I},\;
  \gamma^{\mu\dagger} = \gamma^0\gamma^\mu\gamma^0 \\
  \gamma^5 = i\gamma^0\gamma^1\gamma^2\gamma^3,\;
  \left\{\gamma^5,\gamma^\mu\right\} = 0 \\

  \text{Chiral representation} \\
  \gamma^0 = \sigma_1\otimes\sigma_0,\;
  \gamma^k = -\,i\sigma_2\otimes\sigma_k \\
  \gamma^5 = -\sigma_3\otimes\sigma_0 \\

  \text{Lorentz covariance} \\
  {x'}^\mu = \Lambda^\mu_{\phantom{\mu}\nu}x^\nu,\quad
  \psi' = \mathrm{S}(\Lambda)\psi \\
  \mathrm{S}^{-1}(\Lambda)\gamma^\mu\mathrm{S}(\Lambda) = \Lambda^\mu_{\phantom{\mu}\nu}\gamma^\nu \\
  \mathrm{S}(\Lambda) = \,\mathrm{e}^{\frac{i}{4}\sigma_{\mu\nu}\omega^{\mu\nu}},\;
  \sigma^{\mu\nu} = \frac{i}{2}\left[\gamma^\mu,\gamma^\nu\right] \\
  \mathrm{S}^{-1}(\Lambda) = \gamma^0\mathrm{S}^\dagger(\Lambda)\gamma^0 \\

  \text{Adjoint spinor} \\
  \bar{\psi} = \psi^\dagger\gamma^0,\;
  j^\mu = e\bar{\psi}\gamma^\mu\psi \\
\end{array}
