\begin{array}{l}
  \text{First law + second law} \\
  \begin{array}{lll}
    \theta\mathrm{d}\varsigma & = &
    \mathrm{d}u(\theta,v) + p\mathrm{d}v \\
    & = & \mathrm{d}h(\theta,p) - v\mathrm{d}p
  \end{array} \\

  {\scriptsize
  \dfrac{\partial^2\varsigma}{\partial{\theta}\partial{v}} =
  \dfrac{\partial^2\varsigma}{\partial{v}\partial{\theta}},\quad
  \dfrac{\partial^2\varsigma}{\partial{\theta}\partial{p}} =
  \dfrac{\partial^2\varsigma}{\partial{p}\partial{\theta}} 
  } \\

  \begin{cases}
    \theta\mathrm{d}\varsigma = c_v\mathrm{d}\theta +
    \theta\Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_{v}\mathrm{d}v \\
    \theta\mathrm{d}\varsigma = c_p\mathrm{d}\theta -
    \theta\Big(\dfrac{\partial{v}}{\partial{\theta}}\Big)_{p}\mathrm{d}p
  \end{cases} \\

  \begin{cases}
    \theta\mathrm{d}\varsigma = c_v\mathrm{d}\theta +
    \theta\mathcal{K}_{\theta}\alpha_{p}\mathrm{d}v \\
    \theta\mathrm{d}\varsigma = c_p\mathrm{d}\theta -
    \theta v\alpha_{p}\mathrm{d}p
  \end{cases} \\

  \text{Adiabatic process}\quad \mathrm{d}\varsigma = 0 \\
  \mathcal{K}_{\varsigma} =
  -v\Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_{\varsigma}
  \Big(\dfrac{\partial{\theta}}{\partial{v}}\Big)_{\varsigma} =
  \gamma\mathcal{K}_{\theta} \\

  \text{Heat capacities} \\
  \begin{array}{lll}
    c_p - c_v & = & 
    \theta\Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_{v}
    \Big(\dfrac{\partial{v}}{\partial{\theta}}\Big)_{p} \\
    & = & \theta\mathcal{K}_{\theta}v\alpha_{p}^2
  \end{array}
\end{array}
