\begin{array}{l}
  \text{Bulk modulus} \\
  \mathcal{K}_{x} =
  -v\Big(\dfrac{\partial{p}}{\partial{v}}\Big)_{x}
  \quad x\in\{\theta,\varsigma\} \\

  \text{Compressibility} \; \kappa_{\theta} = \mathcal{K}_{\theta}^{-1} \\

  \text{Isobaric thermal expansion} \\
  \alpha_p = \dfrac{1}{v} \Big(\dfrac{\partial{v}}{\partial{\theta}}\Big)_p \\

  \text{Isochoric thermal expansion} \\
  \beta_v = \dfrac{1}{p} \Big(\dfrac{\partial{p}}{\partial{\theta}}\Big)_v \\

  \text{Heat capacity} \\
  c_{x} = \dfrac{\delta{q}}{\mathrm{d}\theta}\Big\vert_x
  \quad x\in\{v,p\} \\

  \text{First law}\quad\delta{q} = \mathrm{d}u + p\mathrm{d}v\\
  \begin{array}{ll}
    c_v = \Big(\dfrac{\partial{u}}{\partial{\theta}}\Big)_v &
    c_p = \Big(\dfrac{\partial{h}}{\partial{\theta}}\Big)_p
  \end{array} \\

  \text{Enthalpy}\quad h = u + pv
\end{array}
