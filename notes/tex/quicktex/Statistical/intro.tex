\begin{array}{l}
  \text{Ideal gas: average pressure} \\
  p V = N k T, \qquad
  \bar{p}V = N m \bar{v_z^2} \\
  
  \text{Equipartition theorem} \\
  \bar{\mathcal{K}} = \frac{3}{2}N k T, \qquad
  \mathcal{U}_{\theta} = N \cdot f \cdot \frac{1}{2} k T \\
  
  \text{First law} \\
  \Delta\mathcal{U} = \mathcal{Q} + \mathcal {W}, \qquad
  \delta\mathcal{W} = - p \mathrm{d} V \\
\end{array}
